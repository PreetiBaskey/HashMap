class Solution {
public:
    int minimumPushes(string word) {
        int result = 0;
        unordered_map<int, int> mp;

        int assignKey = 2;
        for(char &ch: word) {
            if(assignKey > 9) {
                assignKey = 2;
            }
            mp[assignKey]++;
            result += mp[assignKey];

            assignKey++;
        }

        return result;
    }
};
