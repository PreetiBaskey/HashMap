class Solution {
public:
    int minimumPushes(string word) {
        vector<int> freqArr(26, 0);
        int result = 0;

        for(char &ch: word) {
            freqArr[ch - 'a']++;
        }

        sort(begin(freqArr), end(freqArr), greater<int>());

        for(int i = 0; i < 26; i++) {
            int freq = freqArr[i];
            int press = i/8 + 1;

            result += freq * press;
        }

        return result;
    }
};
