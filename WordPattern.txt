class Solution {
public:
    bool wordPattern(string pattern, string s) {
        vector<string> words;

        stringstream ss(s);

        string token;
        int wordsLength = 0;

        while(getline(ss, token, ' ')) {
            words.push_back(token);
            wordsLength ++;
        }

        if(pattern.length() != wordsLength) {
            return false;
        }

        unordered_map<string, char> mp;
        unordered_set<char> st; 

        for(int i = 0; i < pattern.length(); i++) {
            string str = words[i];
            char ch = pattern[i];

            if(mp.find(str) == mp.end() && st.find(ch) == st.end()) {
                mp[str] = ch;
                st.insert(ch);
            }
            else if(mp[str] != ch) {
                return false;
            }
        }

        return true;
    }
};
